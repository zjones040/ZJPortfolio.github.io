Traffic Light System

In this project, I created a modified traffic system that allows for a longer yellow phase by increasing the yellow interval time by one second. I also explored the benefits of an extended yellow phase on traffic light systems (based on the Virginia Beach traffic light times), formulated a data comparison chart using Excel, and constructed a presentation to explain my project.

int NSGreen = 2; //Attaches north/south green variable to pin two 
int NSYellow = 4; //Attaches north/south yellow variable to pin four 
int NSRed = 6; //Attaches north/south red variable to pin six 
int EWGreen = 3; //Attaches east/west green variable to pin three 
int EWYellow = 5; //Attaches east/west yellow variable to pin five 
int EWRed = 7; //Attaches east/west red variable to pin seven 
void setup() { 
 pinMode ( NSGreen, OUTPUT ); //Sets pin mode for north/south green 
 pinMode ( NSYellow, OUTPUT ); //Sets pin mode for north/south yellow 
 pinMode ( NSRed, OUTPUT ); //Sets pin mode for north/south red 

pinMode ( EWGreen, OUTPUT ); //Sets pin mode for east/west green 
 pinMode ( EWYellow, OUTPUT ); //Sets pin mode for east/west yellow 
 pinMode ( EWRed, OUTPUT ); //Sets pin mode for east/west red 
 //For this circuit setting a pin high turns the LED OFF, and low turns the LED ON 
 digitalWrite ( NSGreen, HIGH ); //Sets default state for north/south green on first pass 
 digitalWrite ( NSYellow, HIGH ); //Sets default state for north/south yellow on first pass 
 digitalWrite ( EWGreen, HIGH ); //Sets default state for east/west green on first pass 
 digitalWrite ( EWYellow, HIGH ); //Sets default state for east/west yellow on first pass 
} 
void loop() { 
 
 digitalWrite ( NSRed, LOW ); //Turn on north/south red 
 digitalWrite ( EWRed, LOW ); //Turn on east/west red 
 delay ( 1020 ); //Wait 1.02 seconds or 1020 milli seconds 
 digitalWrite ( NSRed, HIGH ); //Turn off north/south red 
 digitalWrite ( NSGreen, LOW ); //Turn on north/south green

Arduino Project #2

In this project, I wrote a code using Arduino software that allows the user convert temperatures from Celsius to Fahrenheit and tested it using an Arduino kit. If the converted temperature in Fahrenheit is greater than 90 degrees, then the red LED on the board lights up and the LCD displays the message: “The temp is hot”. If the temperature after the convertion is less than 32 degrees, then the white LED lights up and the LCD displays the message “The temp is cold”. If the temperature in Fahrenheit is between 32 and 90 degrees, then the green LED lights up and the LCD displays the message: “The temp is great”.

Project Code

#include <LiquidCrystal.h>


const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;

LiquidCrystal lcd(rs, en, d4, d5, d6, d7);
 

int pos = 0;

const int ledpin1 = 9;    // LED output pin
const int ledpin2 = 13;
const int ledpin3 = 7;
int Contrast = 60;

void setup(){

Serial.begin(9600);
analogWrite(10, Contrast);
lcd.begin(16, 2);

pinMode(ENABLE, OUTPUT);

pinMode(ledpin1, OUTPUT);
pinMode(ledpin2, OUTPUT);
pinMode(ledpin3, OUTPUT);
}

 

void loop(){

lcd.clear();

int temp = analogRead(A0);



int tempF = map(temp, 0, 1023, 0, 180);  //assuming the input is in F


 //text on first line of LCD

lcd.setCursor(0, 1);

lcd.print("Temp        F"); //text on second line of LCD

lcd.setCursor(6, 1);

lcd.print(tempF); //desply the temp on LCD


if(tempF < 32){  //if temp is below 80F

digitalWrite(ledpin2, HIGH);  //led is ON

digitalWrite(ledpin3, LOW);

digitalWrite(ledpin1, LOW);

lcd.setCursor(0, 0);

lcd.print("Temp is Cold");
delay(200);

}else if(tempF > 90){  //if temp is above 90F

digitalWrite(ledpin1, HIGH);  //led is on

digitalWrite(ledpin2, LOW);

digitalWrite(ledpin3, LOW);


lcd.setCursor(0, 0);

lcd.print("Temp is Hot");
delay(200);

}else{

lcd.setCursor(0, 0);
lcd.print("Temp is great");
digitalWrite(ledpin3, HIGH);

digitalWrite(ledpin2, LOW);

digitalWrite(ledpin1, LOW);  //led is off

delay(200);  //wait 0.1 second

}

}
